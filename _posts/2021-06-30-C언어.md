c언어 기초 학습

# 1.Hello world
```c
#include <stdio.h> //라이브러리 추가하는 부분 , stdio는 기본 라이브러리  

int main(void) //void는 매개 변수이다  
{
	printf("Hello world");
	return 0;
}

```

# 2.size
```c
#include <stdio.h>

int main(void)
{
	int x; 
	x=5;
	printf("변수 x의 메모리 크기는  %d입니다.",sizeof(x)); //sizeof는 메모리크기가 얼마나 되는지 알려줌  
	return 0;
}
```
